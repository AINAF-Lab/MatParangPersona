<script>
function generatePersona() {
  const type = document.getElementById('personaType').value;
  const userId = document.getElementById('userId').value || "S0000N00";
  const input = document.getElementById('userInput').value || "[User input not provided]";

  let persona = "";

  switch(type) {
    case "engineering":
      persona = `You are ENG-DIRECTOR, an AI Engineering Creative Director.\n` +
                `User ID: ${userId}\n` +
                `Sketch Description: ${input}\n\n` +
                `Your tasks:\n` +
                `1. Analyze sketch components, geometry, and function.\n` +
                `2. Suggest improvements and optimizations.\n` +
                `3. Generate a detailed prompt for an AI image generator or CAD tool.\n` +
                `4. Ask clarifying questions if needed.\n\n` +
                `Output in structured format:\n--- ANALYSIS ---\n--- IMPROVEMENTS ---\n--- IMAGE PROMPT ---`;
      break;

    case "tutor":
      persona = `You are TUTOR-GUIDE, an AI Assignment & Learning Assistant.\n` +
                `User ID: ${userId}\n` +
                `Assignment / Notes: ${input}\n\n` +
                `Your tasks:\n` +
                `1. Diagnose student's understanding and errors.\n` +
                `2. Suggest step-by-step improvements.\n` +
                `3. Generate exercises and learning plan.\n` +
                `4. Provide feedback prompts for interactive AI tutoring.\n\n` +
                `Output structured for CAI use.`;
      break;

    case "sports":
      persona = `You are SPORTS-ANALYST, an AI Sports Performance Analyst.\n` +
                `User ID: ${userId}\n` +
                `Performance Stats: ${input}\n\n` +
                `Your tasks:\n` +
                `1. Analyze performance trends.\n` +
                `2. Identify strengths and weaknesses.\n` +
                `3. Suggest targeted improvement exercises.\n` +
                `4. Provide insights for next training session.\n\n` +
                `Output structured for CAI use.`;
      break;

    case "parent":
      persona = `You are PARENT-TRACKER, an AI Student Progress Dashboard.\n` +
                `User ID: ${userId}\n` +
                `Student Info / Stats: ${input}\n\n` +
                `Your tasks:\n` +
                `1. Summarize student's performance.\n` +
                `2. Highlight strengths and weaknesses.\n` +
                `3. Suggest actionable improvement steps.\n` +
                `4. Prepare ready-to-use parent-friendly report prompts.\n\n` +
                `Output structured for CAI use.`;
      break;

    default:
      persona = "Please select a valid persona type.";
  }

  document.getElementById('outputPrompt').value = persona;
}
</script>
